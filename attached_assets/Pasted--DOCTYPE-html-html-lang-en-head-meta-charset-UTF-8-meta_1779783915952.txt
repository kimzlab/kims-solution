<script>

const digitsContainer = document.getElementById("digitsContainer");
const liveTick = document.getElementById("liveTick");
const signalBox = document.getElementById("signalBox");

let digitCounts = Array(10).fill(0);

function generateRandomDigit(){
    return Math.floor(Math.random()*10);
}

function updateAnalysis(){

    let digit = generateRandomDigit();

    digitCounts[digit]++;

    liveTick.innerHTML = `Last Digit: ${digit}`;

    let total = digitCounts.reduce((a,b)=>a+b,0);

    let percentages = digitCounts.map(c => ((c/total)*100).toFixed(1));

    let sorted = [...digitCounts]
    .map((count,index)=>({count,index}))
    .sort((a,b)=>b.count-a.count);

    let top1 = sorted[0].index;
    let top2 = sorted[1].index;
    let top3 = sorted[2].index;

    let low1 = sorted[9].index;
    let low2 = sorted[8].index;
    let low3 = sorted[7].index;

    digitsContainer.innerHTML = "";

    percentages.forEach((p,i)=>{

        let cls = "normal";

        if(i===top1) cls="green";
        else if(i===top2) cls="blue";
        else if(i===top3) cls="lightblue";
        else if(i===low1) cls="red";
        else if(i===low2) cls="orange";
        else if(i===low3) cls="maroon";

        digitsContainer.innerHTML += `
        <div class="digit-box ${cls}">
            <div class="digit">${i}</div>
            <div class="percent">${p}%</div>
        </div>
        `;
    });

    let even = 0;
    let odd = 0;

    digitCounts.forEach((count,index)=>{
        if(index % 2 === 0){
            even += count;
        }else{
            odd += count;
        }
    });

    let evenPercent = ((even/total)*100).toFixed(1);
    let oddPercent = ((odd/total)*100).toFixed(1);

    document.getElementById("evenPercent").innerText = evenPercent + "%";
    document.getElementById("oddPercent").innerText = oddPercent + "%";

    let over = digitCounts.slice(6).reduce((a,b)=>a+b,0);
    let under = digitCounts.slice(0,6).reduce((a,b)=>a+b,0);

    document.getElementById("overPercent").innerText =
    ((over/total)*100).toFixed(1)+"%";

    document.getElementById("underPercent").innerText =
    ((under/total)*100).toFixed(1)+"%";

    if(evenPercent > oddPercent){
        signalBox.innerHTML = "AI SIGNAL → EVEN MARKET STRONG";
    }else{
        signalBox.innerHTML = "AI SIGNAL → ODD MARKET STRONG";
    }

    createCircles(top1);

}

function createCircles(activeDigit){

    let wrapper = document.getElementById("circleWrapper");

    wrapper.innerHTML = "";

    for(let i=0;i<10;i++){

        let active = i===activeDigit ? "active" : "";

        wrapper.innerHTML += `
        <div class="circle ${active}">
            ${i}
        </div>
        `;
    }
}

setInterval(updateAnalysis,1000);

</script>
